    /**
     * Neo4j内部ID
     * 由Neo4j自动生成，不可修改
     */
    @Id
    @GeneratedValue
    private Long id;
    
    /**
     * 业务ID
     * 用于业务逻辑，可手动指定
     */
    private String businessId;
    
    /**
     * 创建时间
     * 使用DateString注解进行序列化
     */
    @DateString
    private LocalDateTime createTime;
    
    /**
     * 更新时间
     */
    @DateString
    private LocalDateTime updateTime;
    
    /**
     * 是否删除（逻辑删除）
     */
    private Boolean deleted;
    
    /**
     * 版本号（乐观锁）
     */
    private Long version;
    
    /**
     * 初始化方法
     */
    public void init() {
        if (this.createTime == null) {
            this.createTime = LocalDateTime.now();
        }
        this.updateTime = LocalDateTime.now();
        this.deleted = false;
        this.version = 0L;
    }
    
    /**
     * 更新方法
     */
    public void update() {
        this.updateTime = LocalDateTime.now();
        if (this.version != null) {
            this.version += 1;
        }
    }
}

@Repository
@RequiredArgsConstructor
public class CustomRouteRepositoryImpl implements CustomRouteRepository {
    
    private final Driver neo4jDriver;
    
    @Override
    public List<RoutePathDTO> findAllRoutes(String startCity, String endCity, int maxDepth) {
        String cypher = "MATCH path = (start:City {name: $startCity})-[*1.." + maxDepth + "]->(end:City {name: $endCity}) " +
                       "WHERE ALL(r IN relationships(path) WHERE r.status = '正常') " +
                       "RETURN path, " +
                       "reduce(distance = 0, r IN relationships(path) | distance + r.distance) AS totalDistance, " +
                       "reduce(cost = 0, r IN relationships(path) | cost + r.cost) AS totalCost, " +
                       "reduce(time = 0, r IN relationships(path) | time + r.time) AS totalTime " +
                       "ORDER BY totalDistance, totalCost, totalTime";
        
        Map<String, Object> parameters = new HashMap<>();
        parameters.put("startCity", startCity);
        parameters.put("endCity", endCity);
        
        try (Session session = neo4jDriver.session()) {
            Result result = session.run(cypher, parameters);
            
            List<RoutePathDTO> routes = new ArrayList<>();
            while (result.hasNext()) {
                Record record = result.next();
                Path path = record.get("path").asPath();
                double totalDistance = record.get("totalDistance").asDouble();
                double totalCost = record.get("totalCost").asDouble();
                int totalTime = record.get("totalTime").asInt();
                
                RoutePathDTO routePath = convertPathToDTO(path);
                routePath.setTotalDistance(totalDistance);
                routePath.setTotalCost(totalCost);
                routePath.setTotalTime(totalTime);
                routePath.setScore(calculateRouteScore(totalDistance, totalCost, totalTime));
                
                routes.add(routePath);
            }
            return routes;
        }
    }
    
    @Override
    public RoutePathDTO findShortestPathByDistance(String startCity, String endCity) {
        String cypher = "MATCH path = shortestPath((start:City {name: $startCity})-[r*..10]-(end:City {name: $endCity})) " +
                       "WHERE ALL(rel IN relationships(path) WHERE rel.distance IS NOT NULL AND rel.status = '正常') " +
                       "RETURN path, " +
                       "reduce(distance = 0, r IN relationships(path) | distance + r.distance) AS totalDistance";
        
        Map<String, Object> parameters = new HashMap<>();
        parameters.put("startCity", startCity);
        parameters.put("endCity", endCity);
        
        try (Session session = neo4jDriver.session()) {
            Result result = session.run(cypher, parameters);
            
            if (result.hasNext()) {
                Record record = result.next();
                Path path = record.get("path").asPath();
                double totalDistance = record.get("totalDistance").asDouble();
                
                RoutePathDTO routePath = convertPathToDTO(path);
                routePath.setTotalDistance(totalDistance);
                routePath.setScore(calculateRouteScore(totalDistance, 0, 0));
                
                return routePath;
            }
            return null;
        }
    }
    
    @Override
    public RoutePathDTO findCheapestRoute(String startCity, String endCity, int maxDepth) {
        String cypher = "MATCH path = (start:City {name: $startCity})-[*1.." + maxDepth + "]->(end:City {name: $endCity}) " +
                       "WHERE ALL(r IN relationships(path) WHERE r.status = '正常') " +
                       "WITH path, " +
                       "reduce(cost = 0, r IN relationships(path) | cost + r.cost) AS totalCost " +
                       "RETURN path, totalCost " +
                       "ORDER BY totalCost ASC " +
                       "LIMIT 1";
        
        Map<String, Object> parameters = new HashMap<>();
        parameters.put("startCity", startCity);
        parameters.put("endCity", endCity);
        
        try (Session session = neo4jDriver.session()) {
            Result result = session.run(cypher, parameters);
            
            if (result.hasNext()) {
                Record record = result.next();
                Path path = record.get("path").asPath();
                double totalCost = record.get("totalCost").asDouble();
                
                RoutePathDTO routePath = convertPathToDTO(path);
                routePath.setTotalCost(totalCost);
                
                // 计算总距离和时间
                double totalDistance = 0;
                int totalTime = 0;
                for (Relationship rel : path.relationships()) {
                    if (rel.containsKey("distance")) {
                        totalDistance += rel.get("distance").asDouble();
                    }
                    if (rel.containsKey("time")) {
                        totalTime += rel.get("time").asInt();
                    }
                }
                routePath.setTotalDistance(totalDistance);
                routePath.setTotalTime(totalTime);
                routePath.setScore(calculateRouteScore(totalDistance, totalCost, totalTime));
                
                return routePath;
            }
            return null;
        }
    }
    
    @Override
    public RoutePathDTO findFastestRoute(String startCity, String endCity, int maxDepth) {
        String cypher = "MATCH path = (start:City {name: $startCity})-[*1.." + maxDepth + "]->(end:City {name: $endCity}) " +
                       "WHERE ALL(r IN relationships(path) WHERE r.status = '正常' AND r.time IS NOT NULL) " +
                       "WITH path, " +
                       "reduce(time = 0, r IN relationships(path) | time + r.time) AS totalTime " +

                double totalDistance = 0;

                        totalCost += rel.get("cost").asDouble();
                    }

                return routePath;
            }
            return null;
        }
    }
    
    @Override
    public RoutePathDTO findOptimalRoute(String startCity, String endCity, 
                                        double distanceWeight, 
                                        double costWeight, 
                                        double timeWeight) {
        String cypher = "MATCH path = (start:City {name: $startCity})-[*1..6]->(end:City {name: $endCity}) " +
                       "WHERE ALL(r IN relationships(path) WHERE r.status = '正常') " +
                       "WITH path, " +
                       "reduce(distance = 0, r IN relationships(path) | distance + r.distance) AS totalDistance, " +
                       "reduce(cost = 0, r IN relationships(path) | cost + r.cost) AS totalCost, " +
                       "reduce(time = 0, r IN relationships(path) | time + r.time) AS totalTime " +
                       "RETURN path, totalDistance, totalCost, totalTime, " +
                       "($distanceWeight * totalDistance + $costWeight * totalCost + $timeWeight * totalTime) AS score " +
                       "ORDER BY score ASC " +
                       "LIMIT 1";
        
        Map<String, Object> parameters = new HashMap<>();
        parameters.put("startCity", startCity);
        parameters.put("endCity", endCity);
        parameters.put("distanceWeight", distanceWeight);
        parameters.put("costWeight", costWeight);
        parameters.put("timeWeight", timeWeight);
        
        try (Session session = neo4jDriver.session()) {
            Result result = session.run(cypher, parameters);
            
            if (result.hasNext()) {
                Record record = result.next();
                Path path = record.get("path").asPath();
                double totalDistance = record.get("totalDistance").asDouble();
                double totalCost = record.get("totalCost").asDouble();
                int totalTime = record.get("totalTime").asInt();
                double score = record.get("score").asDouble();
                
                RoutePathDTO routePath = convertPathToDTO(path);
                routePath.setTotalDistance(totalDistance);
                routePath.setTotalCost(totalCost);
                routePath.setTotalTime(totalTime);
                routePath.setScore(score);
                
                return routePath;
            }
            return null;
        }
    }
    
    /**
     * 将Path转换为RoutePathDTO
     */
    private RoutePathDTO convertPathToDTO(Path path) {
        RoutePathDTO routePath = new RoutePathDTO();
        List<RouteSegmentDTO> segments = new ArrayList<>();
        
        // 处理节点
        List<Node> nodes = new ArrayList<>();
        path.nodes().forEach(nodes::add);
        
        // 处理关系
        List<Relationship> relationships = new ArrayList<>();
        path.relationships().forEach(relationships::add);
        
        // 构建路段
        for (int i = 0; i < relationships.size(); i++) {
            Relationship rel = relationships.get(i);
            Node fromNode = nodes.get(i);
            Node toNode = nodes.get(i + 1);
            
            RouteSegmentDTO segment = new RouteSegmentDTO();
            segment.setFromCity(fromNode.get("name").asString());
            segment.setToCity(toNode.get("name").asString());
            
            if (rel.containsKey("distance")) {
                segment.setDistance(rel.get("distance").asDouble());
            }
            if (rel.containsKey("cost")) {
                segment.setCost(rel.get("cost").asDouble());
            }
            if (rel.containsKey("time")) {
                segment.setTime(rel.get("time").asInt());
            }
            if (rel.containsKey("transport_type")) {
                segment.setTransportType(rel.get("transport_type").asString());
            }
            
            segments.add(segment);
        }
        
        routePath.setSegments(segments);
        routePath.setNodeCount(nodes.size());
        routePath.setSegmentCount(segments.size());
        
        return routePath;
    }
    
    /**
     * 计算路线评分
     */
    private double calculateRouteScore(double distance, double cost, int time) {
        // 简单评分算法：距离 * 0.4 + 成本 * 0.3 + 时间 * 0.3
        return distance * 0.4 + cost * 0.3 + time * 0.3;
    }
}

@Service
@RequiredArgsConstructor
@Slf4j
@Transactional
public class RouteService {
    
    private final CustomRouteRepository customRouteRepository;
    private final CityRepository cityRepository;
    
    /**
     * 查询所有路线
     */
    @Transactional(readOnly = true)
    @Cacheable(value = "routes", key = "#request.toString()")
    public RouteResponseDTO findRoutes(RouteRequestDTO request) {
        log.info("查询路线: {} -> {}, 策略: {}", 
                request.getStartCity(), request.getEndCity(), request.getStrategy());
        
        RouteResponseDTO response = new RouteResponseDTO();
        response.setRequest(request);
        response.setQueryTime(LocalDateTime.now());
        
        List<RoutePathDTO> routes;
        
        switch (request.getStrategy()) {
            case "shortest":
                RoutePathDTO shortestPath = customRouteRepository.findShortestPathByDistance(
                    request.getStartCity(), request.getEndCity());
                routes = shortestPath != null ? List.of(shortestPath) : List.of();
                response.setDescription("最短距离路线");
                break;
                
            case "cheapest":
                RoutePathDTO cheapestRoute = customRouteRepository.findCheapestRoute(
                    request.getStartCity(), request.getEndCity(), request.getMaxDepth());
                routes = cheapestRoute != null ? List.of(cheapestRoute) : List.of();
                response.setDescription("最低成本路线");
                break;
                
            case "fastest":
                RoutePathDTO fastestRoute = customRouteRepository.findFastestRoute(
                    request.getStartCity(), request.getEndCity(), request.getMaxDepth());
                routes = fastestRoute != null ? List.of(fastestRoute) : List.of();
                response.setDescription("最快时间路线");
                break;
                
            case "optimal":
                RoutePathDTO optimalRoute = customRouteRepository.findOptimalRoute(
                    request.getStartCity(), request.getEndCity(),
                    request.getDistanceWeight(), request.getCostWeight(), request.getTimeWeight());
                routes = optimalRoute != null ? List.of(optimalRoute) : List.of();
                response.setDescription("综合最优路线");
                break;
                
            case "all":
            default:
                routes = customRouteRepository.findAllRoutes(
                    request.getStartCity(), request.getEndCity(), request.getMaxDepth());
                response.setDescription("所有可能路线");
                break;
        }
        
        // 排序和过滤
        routes = filterAndSortRoutes(routes, request);
        
        response.setRoutes(routes);
        response.setTotalCount(routes.size());
        
        if (!routes.isEmpty()) {
            response.setBestRoute(routes.get(0));
        }
        
        log.info("查询完成，找到 {} 条路线", routes.size());
        return response;
    }
    
    /**
     * 过滤和排序路线
     */
    private List<RoutePathDTO> filterAndSortRoutes(List<RoutePathDTO> routes, RouteRequestDTO request) {
        return routes.stream()
            .filter(route -> filterRoute(route, request))
            .sorted(getRouteComparator(request.getSortBy()))
            .limit(request.getLimit())
            .collect(Collectors.toList());
    }
    
    /**
     * 路线过滤条件
     */
    private boolean filterRoute(RoutePathDTO route, RouteRequestDTO request) {
        // 过滤最大距离
        if (request.getMaxDistance() != null && route.getTotalDistance() > request.getMaxDistance()) {
            return false;
        }
        
        // 过滤最大成本
        if (request.getMaxCost() != null && route.getTotalCost() > request.getMaxCost()) {
            return false;
        }
        
        // 过滤最大时间
        if (request.getMaxTime() != null && route.getTotalTime() > request.getMaxTime()) {
            return false;
        }
        
        // 过滤最大节点数
        if (request.getMaxNodes() != null && route.getNodeCount() > request.getMaxNodes()) {
            return false;
        }
        
        return true;
    }
    
    /**
     * 获取排序比较器
     */
    private Comparator<RoutePathDTO> getRouteComparator(String sortBy) {
        switch (sortBy) {
            case "distance":
                return Comparator.comparing(RoutePathDTO::getTotalDistance);
            case "cost":
                return Comparator.comparing(RoutePathDTO::getTotalCost);
            case "time":
                return Comparator.comparing(RoutePathDTO::getTotalTime);
            case "score":
                return Comparator.comparing(RoutePathDTO::getScore);
            default:
                return Comparator.comparing(RoutePathDTO::getScore);
        }
    }
    
    /**
     * 添加运输路线
     */
    public void addTransportRoute(String fromCity, String toCity, TransportRoute route) {
        Optional<CityEntity> fromOpt = cityRepository.findByName(fromCity);
        Optional<CityEntity> toOpt = cityRepository.findByName(toCity);
        
        if (fromOpt.isPresent() && toOpt.isPresent()) {
            CityEntity from = fromOpt.get();
            route.setTargetCity(toOpt.get());
            
            from.addRoute(route);
            cityRepository.save(from);
            
            log.info("添加运输路线: {} -> {}, 距离: {}km, 成本: {}元, 时间: {}分钟",
                    fromCity, toCity, route.getDistance(), route.getCost(), route.getTime());
        } else {
            throw new RuntimeException("城市不存在: " + fromCity + " 或 " + toCity);
        }
    }
    
    /**
     * 更新运输路线
     */
    public void updateTransportRoute(String fromCity, String toCity, TransportRoute updatedRoute) {
        Optional<CityEntity> fromOpt = cityRepository.findByName(fromCity);
        
        if (fromOpt.isPresent()) {
            CityEntity from = fromOpt.get();
            
            // 查找并更新路线
            from.getRoutes().stream()
                .filter(route -> route.getTargetCity().getName().equals(toCity))
                .findFirst()
                .ifPresent(route -> {
                    route.setDistance(updatedRoute.getDistance());
                    route.setCost(updatedRoute.getCost());
                    route.setTime(updatedRoute.getTime());
                    route.setTransportType(updatedRoute.getTransportType());
                    route.setStatus(updatedRoute.getStatus());
                    
                    cityRepository.save(from);
                    
                    log.info("更新运输路线: {} -> {}", fromCity, toCity);
                });
        }
    }
    
    /**
     * 删除运输路线
     */
    public void deleteTransportRoute(String fromCity, String toCity) {
        Optional<CityEntity> fromOpt = cityRepository.findByName(fromCity);
        
        if (fromOpt.isPresent()) {
            CityEntity from = fromOpt.get();
            
            boolean removed = from.getRoutes().removeIf(
                route -> route.getTargetCity().getName().equals(toCity));
            
            if (removed) {
                cityRepository.save(from);
                log.info("删除运输路线: {} -> {}", fromCity, toCity);
            }
        }
    }

                stats.setMostConnectedCity(city.getName());
                stats.setMaxRoutes(city.getRoutes().size());
            });
        }
        
        return stats;
    }
