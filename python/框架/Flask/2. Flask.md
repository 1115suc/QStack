## 一、应用工厂模式（Application Factory）

入门时直接 `app = Flask(__name__)` 在模块顶层创建实例，这在大型项目中会带来循环导入等问题。**工厂模式**是 Flask 官方推荐的标准做法。

```python
# app/__init__.py
from flask import Flask
from .config import config_map

def create_app(config_name='development'):
    app = Flask(__name__)
    
    # 加载配置
    app.config.from_object(config_map[config_name])
    
    # 注册扩展
    from .extensions import db, migrate
    db.init_app(app)
    migrate.init_app(app, db)
    
    # 注册蓝图
    from .api.user import user_bp
    from .api.auth import auth_bp
    app.register_blueprint(user_bp, url_prefix='/api/user')
    app.register_blueprint(auth_bp, url_prefix='/api/auth')
    
    return app
```

```python
# run.py（入口文件）
from app import create_app

app = create_app('development')

if __name__ == '__main__':
    app.run()
```

## 二、蓝图（Blueprint）

蓝图是 Flask 的**模块化路由**解决方案，将不同功能的路由拆分到独立文件中，避免所有路由堆在一个文件里。

### 2.1 创建蓝图

```python
# app/api/user.py
from flask import Blueprint, jsonify

user_bp = Blueprint('user', __name__)   # 第一个参数是蓝图名称

@user_bp.route('/', methods=['GET'])
def get_users():
    return jsonify({'users': []})

@user_bp.route('/<int:user_id>', methods=['GET'])
def get_user(user_id):
    return jsonify({'id': user_id})
```

### 2.2 注册蓝图

```python
# 在工厂函数中注册
app.register_blueprint(user_bp, url_prefix='/api/user')
# 访问路径：GET /api/user/
#           GET /api/user/1
```

### 2.3 推荐项目结构

```
my-flask-app/
├── run.py                  # 启动入口
├── config.py               # 配置类
├── requirements.txt
└── app/
    ├── __init__.py         # 工厂函数 create_app()
    ├── extensions.py       # 扩展实例（db, redis 等）
    ├── models/             # 数据模型
    │   ├── __init__.py
    │   └── user.py
    └── api/                # 路由蓝图
        ├── __init__.py
        ├── user.py
        └── auth.py
```

## 三、配置管理

```python
# config.py
import os

class BaseConfig:
    SECRET_KEY = os.environ.get('SECRET_KEY', 'dev-fallback-key')
    JSON_AS_ASCII = False       # 返回 JSON 支持中文
    
class DevelopmentConfig(BaseConfig):
    DEBUG = True
    SQLALCHEMY_DATABASE_URI = 'mysql+pymysql://root:123456@localhost/flask_dev'
    
class ProductionConfig(BaseConfig):
    DEBUG = False
    SQLALCHEMY_DATABASE_URI = os.environ.get('DATABASE_URL')
    
class TestingConfig(BaseConfig):
    TESTING = True
    SQLALCHEMY_DATABASE_URI = 'sqlite:///:memory:'

config_map = {
    'development': DevelopmentConfig,
    'production':  ProductionConfig,
    'testing':     TestingConfig,
}
```

## 四、数据库集成（Flask-SQLAlchemy）

### 4.1 安装


### 4.2 初始化扩展

```python
# app/extensions.py
from flask_sqlalchemy import SQLAlchemy
from flask_migrate import Migrate

db = SQLAlchemy()
migrate = Migrate()
```

### 4.3 定义模型

```python
# app/models/user.py
from app.extensions import db
from datetime import datetime

class User(db.Model):
    __tablename__ = 'users'
    
    id         = db.Column(db.Integer, primary_key=True)
    username   = db.Column(db.String(64), unique=True, nullable=False)
    email      = db.Column(db.String(120), unique=True, nullable=False)
    password   = db.Column(db.String(255), nullable=False)
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
    is_active  = db.Column(db.Boolean, default=True)
    
    def to_dict(self):
        return {
            'id':         self.id,
            'username':   self.username,
            'email':      self.email,
            'created_at': self.created_at.isoformat()
        }
    
    def __repr__(self):
        return f'<User {self.username}>'
```

### 4.4 数据库迁移

```bash
# 初始化迁移目录（只执行一次）
flask db init

# 生成迁移脚本（每次修改模型后执行）
flask db migrate -m "add user table"

# 应用迁移到数据库
flask db upgrade

# 回滚到上一个版本
flask db downgrade
```

### 4.5 CRUD 操作

```python
from app.extensions import db
from app.models.user import User

# 查询
user = User.query.get(1)                          # 按主键查

users = User.query.filter(User.is_active == True).all()

# 新增

db.session.commit()

# 修改

# 删除
user = User.query.get(1)
db.session.delete(user)
db.session.commit()
```

## 五、错误处理

### 5.1 注册错误处理器

```python
from flask import jsonify

@app.errorhandler(404)
def not_found(e):
    return jsonify({'code': 404, 'message': '资源不存在'}), 404

@app.errorhandler(500)
def server_error(e):
    return jsonify({'code': 500, 'message': '服务器内部错误'}), 500
```

### 5.2 自定义异常类（推荐）

```python
# app/exceptions.py
class APIException(Exception):
    code    = 500
    message = '服务器错误'
    
    def __init__(self, message=None, code=None):
        if message:
            self.message = message
        if code:
            self.code = code

class NotFound(APIException):
    code    = 404
    message = '资源不存在'

class Forbidden(APIException):
    code    = 403
    message = '无权访问'

class ParameterError(APIException):
    code    = 400
    message = '参数错误'
```

```python
# 在工厂函数中注册统一处理
@app.errorhandler(APIException)
def handle_api_exception(e):
    return jsonify({'code': e.code, 'message': e.message}), e.code

# 在路由中使用
@user_bp.route('/<int:user_id>')
def get_user(user_id):
    user = User.query.get(user_id)
    if not user:
        raise NotFound('用户不存在')
    return jsonify(user.to_dict())
```

## 六、请求钩子（Hook）

钩子函数在请求的特定阶段自动执行，常用于身份校验、日志记录等。

```python
@app.before_request
def before_request():
    """每个请求处理前执行"""
    print(f'收到请求：{request.method} {request.path}')

@app.after_request
def after_request(response):
    """每个请求处理后执行（即使有异常也会执行）"""
    response.headers['X-Frame-Options'] = 'DENY'
    return response    # 必须返回 response 对象

@app.teardown_appcontext
def teardown(exception):
    """请求上下文销毁时执行，常用于关闭数据库连接"""
    db.session.remove()
```

蓝图级别的钩子（只作用于当前蓝图）：

```python
@user_bp.before_request
def check_auth():
    # 在 user 蓝图所有请求前执行身份校验
    pass
```

```python
from flask_login import LoginManager
login_manager = LoginManager()

from flask_login import UserMixin

class User(db.Model, UserMixin):
    # ... 字段定义

# 加载用户
@login_manager.user_loader
def load_user(user_id):
    return User.query.get(int(user_id))
```

## 八、统一响应格式封装

API 项目通常要统一响应结构，封装一个响应工具函数：

```python
# app/utils/response.py
from flask import jsonify

def success(data=None, message='success', code=200):
    return jsonify({
        'code':    0,
        'message': message,
        'data':    data
    }), code

def fail(message='error', code=400):
    return jsonify({
        'code':    code,
        'message': message,
        'data':    None
    }), code
```

```python
# 使用
from app.utils.response import success, fail

@user_bp.route('/<int:user_id>')
def get_user(user_id):
    user = User.query.get(user_id)
    if not user:
        return fail('用户不存在', 404)
    return success(user.to_dict())
```

## 九、小结

|概念|关键点|
|---|---|
|工厂模式|`create_app()` 统一创建实例，避免循环导入|
|蓝图|按功能模块拆分路由，`url_prefix` 统一前缀|
|数据库|SQLAlchemy ORM + Flask-Migrate 做迁移管理|
|错误处理|自定义异常类 + `@errorhandler` 统一捕获|
|请求钩子|`before_request` / `after_request` 做横切逻辑|
|认证|Flask-Login 管理用户会话|
