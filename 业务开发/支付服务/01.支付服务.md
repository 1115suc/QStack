# 支付宝与微信支付开发集成指南

## 一、支付宝支付集成

### 1. 普通SDK方式

#### 1.1 依赖引入

```xml
<!-- 支付宝普通SDK -->
<dependency>
    <groupId>com.alipay.sdk</groupId>
    <artifactId>alipay-sdk-java</artifactId>
    <version>4.38.100.ALL</version>
</dependency>
<dependency>
    <groupId>commons-logging</groupId>
    <artifactId>commons-logging</artifactId>
    <version>1.2</version>
</dependency>
```

#### 1.2 配置文件

**application.yml/application.properties:**
```yaml
alipay:
  app-id: 你的APP_ID
  merchant-private-key: |
    -----BEGIN PRIVATE KEY-----
    你的商户私钥
    -----END PRIVATE KEY-----
  alipay-public-key: |
    -----BEGIN PUBLIC KEY-----
    支付宝公钥
    -----END PUBLIC KEY-----
  gateway-url: https://openapi.alipay.com/gateway.do
  # 沙箱环境：https://openapi.alipaydev.com/gateway.do
  notify-url: https://your-domain.com/alipay/notify
  return-url: https://your-domain.com/alipay/return
  charset: UTF-8
  sign-type: RSA2
  format: json
  log-path: /tmp/alipay-logs
```

#### 1.4 支付服务实现

```java
@Service
@Slf4j
public class AlipayService {
    
    @Autowired
    private AlipayProperties alipayProperties;
    
    /**
     * 创建支付宝客户端实例
     */
    private AlipayClient createAlipayClient() {
        return new DefaultAlipayClient(
            alipayProperties.getGatewayUrl(),
            alipayProperties.getAppId(),
            alipayProperties.getMerchantPrivateKey(),
            alipayProperties.getFormat(),
            alipayProperties.getCharset(),
            alipayProperties.getAlipayPublicKey(),
            alipayProperties.getSignType()
        );
    }
    
    /**
     * 统一下单并生成支付二维码（面对面支付）
     */
    public String createPreOrder(String outTradeNo, String totalAmount, String subject) {
        AlipayClient alipayClient = createAlipayClient();
        AlipayTradePrecreateRequest request = new AlipayTradePrecreateRequest();
        
        // 设置异步通知地址
        request.setNotifyUrl(alipayProperties.getNotifyUrl());
        
        // 设置业务参数
        JSONObject bizContent = new JSONObject();
        bizContent.put("out_trade_no", outTradeNo);
        bizContent.put("total_amount", totalAmount);
        bizContent.put("subject", subject);
        bizContent.put("timeout_express", "15m");
        
        request.setBizContent(bizContent.toJSONString());
        
        try {
            AlipayTradePrecreateResponse response = alipayClient.execute(request);
            if (response.isSuccess()) {
                return response.getQrCode();
            } else {
                log.error("支付宝预下单失败: {}", response.getSubMsg());
                throw new RuntimeException("支付宝预下单失败: " + response.getSubMsg());
            }
        } catch (AlipayApiException e) {
            log.error("支付宝接口调用异常", e);
            throw new RuntimeException("支付宝接口调用异常", e);
        }
    }
    
    /**
     * 查询订单状态
     */
    public AlipayTradeQueryResponse queryOrder(String outTradeNo) {
        AlipayClient alipayClient = createAlipayClient();
        AlipayTradeQueryRequest request = new AlipayTradeQueryRequest();
        
        JSONObject bizContent = new JSONObject();
        bizContent.put("out_trade_no", outTradeNo);
        request.setBizContent(bizContent.toJSONString());
        
        try {
            return alipayClient.execute(request);
        } catch (AlipayApiException e) {
            log.error("支付宝查询订单异常", e);
            throw new RuntimeException("支付宝查询订单异常", e);
        }
    }
    
    /**
     * 退款
     */
    public AlipayTradeRefundResponse refund(String outTradeNo, String refundAmount) {
        AlipayClient alipayClient = createAlipayClient();
        AlipayTradeRefundRequest request = new AlipayTradeRefundRequest();
        
        JSONObject bizContent = new JSONObject();
        bizContent.put("out_trade_no", outTradeNo);
        bizContent.put("refund_amount", refundAmount);
        bizContent.put("refund_reason", "正常退款");
        request.setBizContent(bizContent.toJSONString());
        
        try {
            return alipayClient.execute(request);
        } catch (AlipayApiException e) {
            log.error("支付宝退款异常", e);
            throw new RuntimeException("支付宝退款异常", e);
        }
    }
    
    /**
     * 验证签名（用于回调验证）
     */
    public boolean verifySignature(Map<String, String> params) {
        try {
            return AlipaySignature.rsaCheckV1(
                params,
                alipayProperties.getAlipayPublicKey(),
                alipayProperties.getCharset(),
                alipayProperties.getSignType()
            );
        } catch (AlipayApiException e) {
            log.error("支付宝签名验证异常", e);
            return false;
        }
    }
}
```

#### 2.2 配置文件

```yaml
alipay:
  easy:
    app-id: 你的APP_ID
    protocol: https
    gateway-host: openapi.alipay.com
    # 沙箱环境：openapi.alipaydev.com
    sign-type: RSA2
    merchant-private-key: |
      -----BEGIN PRIVATE KEY-----
      你的商户私钥
      -----END PRIVATE KEY-----
    alipay-public-key: |
      -----BEGIN PUBLIC KEY-----
      支付宝公钥
      -----END PUBLIC KEY-----
    notify-url: https://your-domain.com/alipay/notify
    encrypt-key: 你的AES密钥  # 可选
```

#### 2.3 配置类

```java
@Configuration
public class AlipayEasyConfig {
    
    @Bean
    @ConfigurationProperties(prefix = "alipay.easy")
    public AlipayEasyProperties alipayEasyProperties() {
        return new AlipayEasyProperties();
    }
    
    @Bean
    public Config alipayConfig(AlipayEasyProperties properties) {
        Config config = new Config();
        config.protocol = properties.getProtocol();
        config.gatewayHost = properties.getGatewayHost();
        config.signType = properties.getSignType();
        config.appId = properties.getAppId();
        config.merchantPrivateKey = properties.getMerchantPrivateKey();
        config.alipayPublicKey = properties.getAlipayPublicKey();
        config.notifyUrl = properties.getNotifyUrl();
        
        if (StringUtils.isNotBlank(properties.getEncryptKey())) {
            config.encryptKey = properties.getEncryptKey();
        }
        
        return config;
    }
    
    @Bean
    public void initFactory(Config config) {
        Factory.setOptions(config);
    }
}
```

```java
                .merchantId(properties.getMchId())
                .privateKey(merchantPrivateKey)
                .merchantSerialNumber(properties.getMerchantSerialNo())
                .apiV3Key(properties.getApiV3Key())
                .build();

                .build();
            
            refundService = new RefundService.Builder()
                .config(config)
                .build();

    private PrivateKey getPrivateKey(String privateKeyContent) throws Exception {
        String privateKey = privateKeyContent
            .replace("-----BEGIN PRIVATE KEY-----", "")
            .replace("-----END PRIVATE KEY-----", "")
            .replaceAll("\\s+", "");
        
        byte[] keyBytes = Base64.getDecoder().decode(privateKey);
        PKCS8EncodedKeySpec keySpec = new PKCS8EncodedKeySpec(keyBytes);
        KeyFactory keyFactory = KeyFactory.getInstance("RSA");
        return keyFactory.generatePrivate(keySpec);
    }

            request.setMchid(properties.getMchId());
            request.setDescription(description);
            request.setOutTradeNo(outTradeNo);
            request.setAmount(orderAmount);
            request.setNotifyUrl(properties.getNotifyUrl());

        }
```

### 2. 支付结果处理器

```java
@Service
@Slf4j
public class PaymentResultHandler {
    
    @Autowired
    private TradingService tradingService;
    
    /**
     * 处理支付宝支付回调
     */
    @Transactional
    public void handleAlipayCallback(Map<String, String> params) {
        try {
            String outTradeNo = params.get("out_trade_no");
            String tradeNo = params.get("trade_no");
            String tradeStatus = params.get("trade_status");
            String totalAmount = params.get("total_amount");
            
            Trading trading = tradingService.findByTradeNo(outTradeNo);
            if (trading == null) {
                log.error("订单不存在: {}", outTradeNo);
                return;
            }
            
            switch (tradeStatus) {
                case "TRADE_SUCCESS":
                case "TRADE_FINISHED":
                    tradingService.updateTradeStatus(outTradeNo, "YJS", tradeNo, totalAmount);
                    break;
                case "TRADE_CLOSED":
                    tradingService.updateTradeStatus(outTradeNo, "FKSB");
                    break;
                default:
                    log.warn("未知的支付宝交易状态: {}", tradeStatus);
            }
        } catch (Exception e) {
            log.error("处理支付宝回调异常", e);
            throw new RuntimeException("处理支付回调失败", e);
        }
    }
    
    /**
     * 处理微信支付回调
     */
    @Transactional
    public void handleWechatCallback(Map<String, String> params) {
        try {
            String outTradeNo = params.get("out_trade_no");
            String transactionId = params.get("transaction_id");
            String returnCode = params.get("return_code");
            String resultCode = params.get("result_code");
            
            if ("SUCCESS".equals(returnCode) && "SUCCESS".equals(resultCode)) {
                String totalFee = params.get("total_fee");
                tradingService.updateTradeStatus(outTradeNo, "YJS", transactionId, 
                    new BigDecimal(totalFee).divide(new BigDecimal("100")).toString());
            } else {
                tradingService.updateTradeStatus(outTradeNo, "FKSB");
            }
        } catch (Exception e) {
            log.error("处理微信回调异常", e);
            throw new RuntimeException("处理支付回调失败", e);
        }
    }
```
