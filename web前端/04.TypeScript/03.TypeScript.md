## 一、模块系统

### 1.1 ES Module 语法

TypeScript 完全支持 ES Module：

```typescript
// math.ts —— 导出
export const PI = 3.14159;

export function add(a: number, b: number): number {
  return a + b;
}

export default class Calculator {
  multiply(a: number, b: number): number {
    return a * b;
  }
}

// main.ts —— 导入
import Calculator, { PI, add } from "./math";
import * as MathUtils from "./math"; // 命名空间导入
```

### 1.2 类型的导入与导出

```typescript
// 导出类型（TS 3.8+ 的 type-only 导出）
export type { User, Config };
export type UserID = string | number;

// 仅导入类型（编译后会被完全擦除，不产生运行时代码）
import type { User } from "./types";
```

> **推荐**：使用 `import type` 导入类型，明确区分值导入和类型导入，有助于 Tree Shaking。

### 1.3 动态导入

```typescript
// 动态导入（返回 Promise）
async function loadModule() {
  const { default: Calculator } = await import("./math");
  const calc = new Calculator();
  return calc.multiply(2, 3);
}

// 类型安全的动态导入
type MathModule = typeof import("./math");
```

## 二、声明文件（.d.ts）

### 2.1 什么是声明文件？

声明文件只包含**类型信息**，不包含实际实现，用于为 JavaScript 库提供类型支持：

```typescript
// jquery.d.ts（简化示例）
declare const $: (selector: string) => {
  show(): void;
  hide(): void;
  html(content?: string): string | void;
};
```

### 2.2 声明的种类

```typescript
// 声明变量
declare const VERSION: string;
declare let count: number;

// 声明函数
declare function ajax(url: string, options?: object): Promise<any>;

// 声明类
declare class EventEmitter {
  on(event: string, listener: Function): this;
  off(event: string, listener: Function): this;
  emit(event: string, ...args: any[]): boolean;
}

// 声明枚举
declare enum Status {
  Active,
  Inactive,
}

// 声明模块
declare module "lodash" {
  export function chunk<T>(array: T[], size: number): T[][];
  export function debounce<T extends Function>(fn: T, wait: number): T;
}

// 声明命名空间
declare namespace MyLib {
  function create(options: object): void;
  interface Options {
    timeout: number;
  }
}
```

### 2.3 安装第三方类型声明

```bash
# 大多数流行库的类型声明由 DefinitelyTyped 社区维护
npm install --save-dev @types/lodash
npm install --save-dev @types/node
npm install --save-dev @types/react

# 查询是否有对应 @types 包
npx typesync  # 自动添加缺少的 @types
```

### 2.4 扩展已有模块（Module Augmentation）

```typescript
// 扩展 Express 的 Request 类型
import "express";

declare module "express" {
  interface Request {
    user?: {
      id: number;
      name: string;
    };
  }
}

// 扩展全局类型
declare global {
  interface Window {
    __APP_VERSION__: string;
  }
  interface Array<T> {
    last(): T | undefined;
  }
}

Array.prototype.last = function () {
  return this[this.length - 1];
};
```

## 三、命名空间（Namespace）

命名空间用于组织代码，避免全局污染（在模块化盛行的今天，多在声明文件中使用）：

```typescript
namespace Validation {
  // 内部接口
  export interface StringValidator {
    isAcceptable(s: string): boolean;
  }

  const lettersRegexp = /^[A-Za-z]+$/;

  // 导出类
  export class LettersOnlyValidator implements StringValidator {
    isAcceptable(s: string): boolean {
      return lettersRegexp.test(s);
    }
  }
}

// 使用
const validator = new Validation.LettersOnlyValidator();
console.log(validator.isAcceptable("Hello")); // true

// 命名空间嵌套
namespace App.Utils {
  export function formatDate(date: Date): string {
    return date.toISOString().split("T")[0];
  }
}
```

## 四、tsconfig.json 深度配置

### 4.1 常用配置详解

```json
{
  "compilerOptions": {
    /* 基础选项 */
    "target": "ES2020",          // 编译目标：ES5, ES6, ES2020, ESNext
    "lib": ["ES2020", "DOM"],    // 内置类型库
    "module": "ESNext",          // 模块规范：CommonJS, ESNext, AMD
    "moduleResolution": "bundler", // 模块解析：node, bundler (TS 5.0+)
    "outDir": "./dist",
    "rootDir": "./src",

    /* 严格模式 */
    "strict": true,              // 开启所有严格检查（推荐）
    "noImplicitAny": true,       // 不允许隐式 any
    "strictNullChecks": true,    // 不允许 null/undefined 赋给其他类型
    "strictFunctionTypes": true, // 严格函数参数类型检查
    "noImplicitReturns": true,   // 函数所有分支都要有返回值

    /* 代码质量 */
    "noUnusedLocals": true,      // 不允许未使用的局部变量
    "noUnusedParameters": true,  // 不允许未使用的参数
    "exactOptionalPropertyTypes": true, // 可选属性不能赋 undefined（TS 4.4+）

    /* 路径别名 */
    "baseUrl": ".",
    "paths": {
      "@/*": ["src/*"],
      "@components/*": ["src/components/*"]
    },

    /* 声明文件 */
    "declaration": true,         // 生成 .d.ts 文件
    "declarationMap": true,      // 生成声明文件的 source map

    /* 其他 */
    "sourceMap": true,
    "experimentalDecorators": true,
    "emitDecoratorMetadata": true,
    "esModuleInterop": true,
    "allowSyntheticDefaultImports": true,
    "skipLibCheck": true         // 跳过 node_modules 中的类型检查（加快编译）
  },
  "include": ["src/**/*"],
  "exclude": ["node_modules", "dist"]
}
```

### 4.2 多环境配置继承

```json
// tsconfig.base.json（公共配置）
{
  "compilerOptions": {
    "strict": true,
    "esModuleInterop": true
  }
}

// tsconfig.json（开发）
{
  "extends": "./tsconfig.base.json",
  "compilerOptions": {
    "sourceMap": true,
    "outDir": "./dist"
  }
}

// tsconfig.prod.json（生产）
{
  "extends": "./tsconfig.base.json",
  "compilerOptions": {
    "sourceMap": false,
    "removeComments": true,
    "outDir": "./dist"
  }
}
```

## 五、与主流框架集成

### 5.1 TypeScript + React

```tsx
import React, { useState, useEffect, FC, ReactNode } from "react";

// 组件 Props 类型
interface ButtonProps {
  label: string;
  onClick: () => void;
  disabled?: boolean;
  variant?: "primary" | "secondary" | "danger";
  children?: ReactNode;
}

// 函数组件
const Button: FC<ButtonProps> = ({
  label,
  onClick,
  disabled = false,
  variant = "primary",
  children,
}) => {
  return (
    <button
      className={`btn btn-${variant}`}
      onClick={onClick}
      disabled={disabled}
    >
      {label}
      {children}
    </button>
  );
};

// useState 类型推断
const [count, setCount] = useState(0);              // 自动推断 number
const [user, setUser] = useState<User | null>(null); // 显式类型参数

// useRef
const inputRef = React.useRef<HTMLInputElement>(null);

// 事件类型
const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
  console.log(e.target.value);
};

const handleSubmit = (e: React.FormEvent<HTMLFormElement>) => {
  e.preventDefault();
};

// 自定义 Hook
function useFetch<T>(url: string): { data: T | null; loading: boolean; error: string | null } {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    fetch(url)
      .then((res) => res.json())
      .then((json: T) => {
        setData(json);
        setLoading(false);
      })
      .catch((err: Error) => {
        setError(err.message);
        setLoading(false);
      });
  }, [url]);

  return { data, loading, error };
}
```

## 六、类型体操（Type-Level Programming）

类型体操是指在**类型层面**进行复杂的逻辑运算，是 TS 高级用法的精髓：

### 6.1 DeepPartial（深度可选）

```typescript
type DeepPartial<T> = T extends object
  ? { [P in keyof T]?: DeepPartial<T[P]> }
  : T;

interface Config {
  server: {
    host: string;
    port: number;
  };
  database: {
    url: string;
    name: string;
  };
}

type PartialConfig = DeepPartial<Config>;
// 所有嵌套属性都变为可选
```

### 6.2 FlattenArray（展开数组）

```typescript
type FlattenArray<T> = T extends Array<infer Item> ? Item : T;

type A = FlattenArray<number[]>;    // number
type B = FlattenArray<string[][]>;  // string[]
type C = FlattenArray<boolean>;     // boolean
```

### 6.3 UnionToIntersection（联合转交叉）

```typescript
type UnionToIntersection<U> =
  (U extends any ? (k: U) => void : never) extends (k: infer I) => void
    ? I
    : never;

type Union = { a: string } | { b: number };
type Intersection = UnionToIntersection<Union>; // { a: string } & { b: number }
```

### 6.4 路径访问类型

```typescript
type PathValue<T, P extends string> =
  P extends `${infer Key}.${infer Rest}`
    ? Key extends keyof T
      ? PathValue<T[Key], Rest>
      : never
    : P extends keyof T
    ? T[P]
    : never;

interface Data {
  user: {
    profile: {
      name: string;
    };
  };
}

type Name = PathValue<Data, "user.profile.name">; // string
```

## 七、最佳实践与常见陷阱

### 7.1 开启严格模式

```json
// tsconfig.json
{
  "compilerOptions": {
    "strict": true  // 始终开启！
  }
}
```

### 7.2 避免 any，优先用 unknown

```typescript
// ❌ 不推荐
function parseData(data: any) {
  return data.name; // 类型不安全，运行时可能报错
}

// ✅ 推荐：使用 unknown + 类型收窄
function parseData(data: unknown) {
  if (typeof data === "object" && data !== null && "name" in data) {
    return (data as { name: string }).name;
  }
  throw new Error("Invalid data");
}
```

### 7.3 使用判别联合（Discriminated Unions）

```typescript
// ✅ 每个分支有唯一的字面量属性（判别式）
type Result<T> =
  | { status: "success"; data: T }
  | { status: "error"; error: string }
  | { status: "loading" };

function handleResult<T>(result: Result<T>) {
  switch (result.status) {
    case "success":
      console.log(result.data); // TS 知道这里有 data
      break;
    case "error":
      console.error(result.error); // TS 知道这里有 error
      break;
    case "loading":
      console.log("Loading...");
      break;
  }
}
```

### 7.4 善用 satisfies 操作符（TS 4.9+）

```typescript
const palette = {
  red: [255, 0, 0],
  green: "#00ff00",
  blue: [0, 0, 255],
} satisfies Record<string, string | number[]>;

// satisfies 在验证类型的同时保留字面量类型
// palette.red 的类型是 number[]（而不是 string | number[]）
palette.red.map(v => v * 2); // ✅ 正确推断为 number[]
```

### 7.5 const 断言

```typescript
// 普通对象：属性类型被推断为宽泛类型
const config1 = { host: "localhost", port: 3000 };
// config1.host 类型为 string

// as const：所有属性变为字面量类型（只读）
const config2 = { host: "localhost", port: 3000 } as const;
// config2.host 类型为 "localhost"
// config2.port 类型为 3000

// 数组 as const：变为只读元组
const routes = ["/home", "/about", "/contact"] as const;
type Route = typeof routes[number]; // "/home" | "/about" | "/contact"
```

### 7.6 常见陷阱

```typescript
// 陷阱1：枚举的数字枚举不是类型安全的
enum Fruit { Apple, Banana }
const f: Fruit = 99; // ✅ 编译不报错（数字枚举允许任意数字）
// 解决：使用字符串枚举或联合类型

// 陷阱2：类型断言不做运行时检查
const x = {} as { name: string };
console.log(x.name.length); // ❌ 运行时报错：Cannot read properties of undefined

// 陷阱3：函数参数双变（strictFunctionTypes 解决）
type Fn = (a: string) => void;
const fn: Fn = (a: string | number) => {}; // ❌ 严格模式下报错

// 陷阱4：对象字面量的多余属性检查
interface Point { x: number; y: number; }
const p: Point = { x: 1, y: 2, z: 3 }; // ❌ 多余属性检查
const obj = { x: 1, y: 2, z: 3 };
const p2: Point = obj; // ✅ 通过（赋值兼容性不检查多余属性）
```

## 八、性能优化与工程化

### 8.1 加快编译速度

```json
{
  "compilerOptions": {
    "skipLibCheck": true,       // 跳过 .d.ts 类型检查
    "incremental": true,        // 增量编译（缓存编译结果）
    "tsBuildInfoFile": ".tsbuildinfo"
  }
}
```

### 8.2 使用 Project References（大型项目）

```json
// packages/core/tsconfig.json
{
  "compilerOptions": {
    "composite": true,
    "outDir": "dist"
  }
}

// tsconfig.json（根目录）
{
  "references": [
    { "path": "./packages/core" },
    { "path": "./packages/ui" }
  ]
}
```
