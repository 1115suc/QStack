## 一、泛型（Generics）

### 1.1 为什么需要泛型？

```typescript
// ❌ 不用泛型：只能处理 number，复用性差
function identity(arg: number): number {
  return arg;
}

// ❌ 用 any：失去类型安全
function identity(arg: any): any {
  return arg;
}

// ✅ 泛型：类型安全 + 可复用
function identity<T>(arg: T): T {
  return arg;
}

identity<string>("hello"); // 显式传入类型参数
identity(42);              // 类型自动推断为 number
```

### 1.2 泛型约束

用 `extends` 限制泛型的类型范围：

```typescript
interface HasLength {
  length: number;
}

// T 必须有 length 属性
function logLength<T extends HasLength>(arg: T): T {
  console.log(arg.length);
  return arg;
}

logLength("hello");       // ✅ string 有 length
logLength([1, 2, 3]);     // ✅ array 有 length
// logLength(42);         // ❌ number 没有 length
```

### 1.3 泛型接口与泛型类

```typescript
// 泛型接口
interface Repository<T> {
  findById(id: number): T;
  findAll(): T[];
  save(entity: T): void;
}

// 泛型类
class Stack<T> {
  private items: T[] = [];

  push(item: T): void {
    this.items.push(item);
  }

  pop(): T | undefined {
    return this.items.pop();
  }

  peek(): T | undefined {
    return this.items[this.items.length - 1];
  }

  get size(): number {
    return this.items.length;
  }
}

const stack = new Stack<number>();
stack.push(1);
stack.push(2);
console.log(stack.pop()); // 2
```

### 1.4 多个类型参数

```typescript
function pair<K, V>(key: K, value: V): [K, V] {
  return [key, value];
}

const result = pair("name", "Alice"); // [string, string]
```

### 1.5 keyof 与泛型

```typescript
// 获取对象属性（类型安全的属性访问）
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}

const user = { name: "Alice", age: 25 };
getProperty(user, "name"); // ✅ string
getProperty(user, "age");  // ✅ number
// getProperty(user, "email"); // ❌ 报错
```

## 二、类（Class）

### 2.1 类的基本写法

```typescript
class Animal {
  // 属性声明（TS 需要先声明再使用）
  name: string;
  private age: number;
  protected species: string;
  readonly id: number;

  constructor(name: string, age: number) {
    this.name = name;
    this.age = age;
    this.species = "Unknown";
    this.id = Math.random();
  }

  speak(): string {
    return `${this.name} makes a sound.`;
  }
}
```

### 2.2 访问修饰符

| 修饰符      | 当前类 | 子类 | 外部 |
| ----------- | ------ | ---- | ---- |
| `public`    | ✅     | ✅   | ✅   |
| `protected` | ✅     | ✅   | ❌   |
| `private`   | ✅     | ❌   | ❌   |

```typescript
// 构造函数参数简写（参数属性）
class Person {
  constructor(
    public name: string,
    private age: number,
    protected email: string
  ) {}
}

// 等价于在构造函数中手动赋值
```

### 2.3 继承与多态

```typescript
class Dog extends Animal {
  constructor(name: string, age: number, public breed: string) {
    super(name, age); // 必须调用 super
  }

  // 方法重写
  speak(): string {
    return `${this.name} barks!`;
  }
}

const dog = new Dog("Rex", 3, "Labrador");
console.log(dog.speak()); // "Rex barks!"
```

### 2.4 抽象类

```typescript
abstract class Shape {
  abstract area(): number; // 抽象方法：子类必须实现

  describe(): string {
    return `This shape has area: ${this.area()}`;
  }
}

class Circle extends Shape {
  constructor(private radius: number) {
    super();
  }

  area(): number {
    return Math.PI * this.radius ** 2;
  }
}

class Rectangle extends Shape {
  constructor(private width: number, private height: number) {
    super();
  }

  area(): number {
    return this.width * this.height;
  }
}

// const shape = new Shape(); // ❌ 抽象类不能实例化
const circle = new Circle(5);
console.log(circle.describe()); // "This shape has area: 78.53..."
```

### 2.5 类实现接口

```typescript
interface Flyable {
  fly(): void;
  altitude: number;
}

interface Swimmable {
  swim(): void;
}

// 一个类可以实现多个接口
class Duck implements Flyable, Swimmable {
  altitude: number = 100;

  fly(): void {
    console.log("Duck is flying");
  }

  swim(): void {
    console.log("Duck is swimming");
  }
}
```

### 2.6 静态成员

```typescript
class MathUtils {
  static PI = 3.14159;

  static circleArea(r: number): number {
    return MathUtils.PI * r * r;
  }
}

console.log(MathUtils.PI);
console.log(MathUtils.circleArea(5));
```

### 2.7 Getter / Setter

```typescript
class Temperature {
  private _celsius: number = 0;

  get fahrenheit(): number {
    return this._celsius * 9/5 + 32;
  }

  set fahrenheit(value: number) {
    this._celsius = (value - 32) * 5/9;
  }

  get celsius(): number {
    return this._celsius;
  }

  set celsius(value: number) {
    if (value < -273.15) throw new Error("Temperature below absolute zero!");
    this._celsius = value;
  }
}

const temp = new Temperature();
temp.celsius = 100;
console.log(temp.fahrenheit); // 212
```

## 三、高级类型

### 3.1 联合类型（Union Types）

```typescript
type StringOrNumber = string | number;

function format(value: StringOrNumber): string {
  if (typeof value === "string") {
    return value.toUpperCase();
  }
  return value.toFixed(2);
}

// 字面量联合类型（枚举的替代方案）
type Direction = "up" | "down" | "left" | "right";
type HttpMethod = "GET" | "POST" | "PUT" | "DELETE";
```

### 3.2 交叉类型（Intersection Types）

```typescript
type Named = { name: string };
type Aged = { age: number };
type Person = Named & Aged; // 同时拥有两者的属性

const person: Person = { name: "Alice", age: 25 };

// 常用于混入（Mixin）模式
function merge<T, U>(a: T, b: U): T & U {
  return { ...a, ...b } as T & U;
}
```

### 3.3 类型别名（Type Alias）

```typescript
// 为复杂类型起名
type ID = string | number;
type Callback = (err: Error | null, result: string) => void;
type Nullable<T> = T | null;
type Optional<T> = T | undefined;

// 递归类型
type TreeNode<T> = {
  value: T;
  children?: TreeNode<T>[];
};
```

### 3.4 映射类型（Mapped Types）

基于已有类型，变换生成新类型：

```typescript
type Original = {
  name: string;
  age: number;
  email: string;
};

// 将所有属性变为可选
type Partial<T> = {
  [P in keyof T]?: T[P];
};

// 将所有属性变为只读
type Readonly<T> = {
  readonly [P in keyof T]: T[P];
};

// 将所有属性变为 string 类型
type Stringify<T> = {
  [P in keyof T]: string;
};

// 修改修饰符（+ 添加，- 移除）
type Mutable<T> = {
  -readonly [P in keyof T]: T[P]; // 去掉 readonly
};
```

### 3.5 条件类型（Conditional Types）

```typescript
// T extends U ? X : Y
type IsString<T> = T extends string ? true : false;

type A = IsString<string>;  // true
type B = IsString<number>;  // false

// 条件类型 + infer（类型推断）
type ReturnType<T> = T extends (...args: any[]) => infer R ? R : never;

function fetchUser(): { name: string; age: number } {
  return { name: "Alice", age: 25 };
}

type UserType = ReturnType<typeof fetchUser>; // { name: string; age: number }

// 提取 Promise 的内部类型
type Awaited<T> = T extends Promise<infer U> ? U : T;
type Inner = Awaited<Promise<string>>; // string
```

### 3.6 模板字面量类型（TS 4.1+）

```typescript
type Greeting = `Hello, ${string}!`;

type EventName = "click" | "focus" | "blur";
type EventHandler = `on${Capitalize<EventName>}`; // "onClick" | "onFocus" | "onBlur"

// 实际应用：生成 CSS 属性
type Direction = "top" | "right" | "bottom" | "left";
type Padding = `padding-${Direction}`; // "padding-top" | ...
```

## 四、内置工具类型

TypeScript 内置了大量工具类型，开箱即用：

### 4.1 对象相关

```typescript
interface User {
  id: number;
  name: string;
  email: string;
  age?: number;
}

// Partial<T>：全部属性变可选
type PartialUser = Partial<User>;

// Required<T>：全部属性变必填
type RequiredUser = Required<User>;

// Readonly<T>：全部属性变只读
type ReadonlyUser = Readonly<User>;

// Pick<T, K>：挑选部分属性
type UserPreview = Pick<User, "id" | "name">;

// Omit<T, K>：排除部分属性
type UserWithoutId = Omit<User, "id">;

// Record<K, V>：构造键值对类型
type RoleMap = Record<string, User>;

// 组合使用
type UpdateUserDto = Partial<Omit<User, "id">>;
```

### 4.2 联合类型相关

```typescript
type T = string | number | boolean;

// Exclude<T, U>：从 T 中排除 U
type NoBoolean = Exclude<T, boolean>; // string | number

// Extract<T, U>：提取 T 中属于 U 的类型
type OnlyString = Extract<T, string>; // string

// NonNullable<T>：排除 null 和 undefined
type NotNull = NonNullable<string | null | undefined>; // string
```

### 4.3 函数相关

```typescript
function greet(name: string, age: number): string {
  return `Hello ${name}, you are ${age}`;
}

// ReturnType<T>：获取函数返回值类型
type GreetReturn = ReturnType<typeof greet>; // string

// Parameters<T>：获取函数参数类型（元组）
type GreetParams = Parameters<typeof greet>; // [name: string, age: number]

// ConstructorParameters<T>：构造函数参数类型
class MyClass {
  constructor(public x: number, public y: number) {}
}
type MyParams = ConstructorParameters<typeof MyClass>; // [x: number, y: number]

// InstanceType<T>：获取类的实例类型
type MyInstance = InstanceType<typeof MyClass>; // MyClass
```

## 五、装饰器（Decorators）

> 装饰器是 TS 的实验性特性，需在 tsconfig 中开启 `"experimentalDecorators": true`

### 5.1 类装饰器

```typescript
function sealed(constructor: Function) {
  Object.seal(constructor);
  Object.seal(constructor.prototype);
}

@sealed
class BugReport {
  type = "report";
  title: string;

  constructor(t: string) {
    this.title = t;
  }
}
```

### 5.2 方法装饰器

```typescript
function log(
  target: any,
  propertyName: string,
  descriptor: PropertyDescriptor
) {
  const original = descriptor.value;
  descriptor.value = function (...args: any[]) {
    console.log(`Calling ${propertyName} with`, args);
    const result = original.apply(this, args);
    console.log(`Result:`, result);
    return result;
  };
}

class Calculator {
  @log
  add(a: number, b: number): number {
    return a + b;
  }
}

const calc = new Calculator();
calc.add(1, 2);
// Calling add with [1, 2]
// Result: 3
```

### 5.3 属性装饰器

```typescript
function readonly(target: any, propertyKey: string) {
  Object.defineProperty(target, propertyKey, {
    writable: false,
  });
}

class User {
  @readonly
  role: string = "user";
}
```

## 总结

| 知识点      | 核心要点                                           |
| ----------- | -------------------------------------------------- |
| 泛型        | 类型参数化，约束用 extends，配合 keyof 使用        |
| 类          | 访问修饰符、继承、抽象类、接口实现                 |
| 联合/交叉   | `|` 表示"或"，`&` 表示"且"                         |
| 映射类型    | `[P in keyof T]` 遍历键，可增删修饰符              |
| 条件类型    | `T extends U ? X : Y`，配合 infer 推断内部类型     |
| 工具类型    | Partial / Required / Pick / Omit / Record 等       |
| 装饰器      | 类/方法/属性装饰器，需开启 experimentalDecorators |
