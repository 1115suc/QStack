# JavaScript 进阶笔记

# 作用域与函数进阶

## 一、作用域

### 1.1 作用域基础

**定义**: 作用域规定了变量能够被访问的"范围",离开这个范围变量便不能被访问

**作用域分类**:
- 全局作用域
- 局部作用域
    - 函数作用域
    - 块作用域

### 1.2 局部作用域

#### 1.2.1 函数作用域

**定义**: 在函数内部声明的变量只能在函数内部被访问,外部无法直接访问

```javascript
// 函数作用域示例
function counter(x, y) {
    const s = x + y
    console.log(s)  // 18
}

counter(10, 8)

// ❌ 错误: 外部无法访问
console.log(s)  // 报错: s is not defined
```

**函数作用域的特点**:

```javascript
// 1. 函数内部变量外部无法访问
function test1() {
    let name = '小明'
    console.log(name)
}
test1()
// console.log(name)  // 报错

// 2. 函数参数也是局部变量
function greet(username) {
    console.log(username)
}
greet('小红')
// console.log(username)  // 报错

// 3. 不同函数内部变量互不影响
function fn1() {
    let num = 10
}

function fn2() {
    let num = 20
}

// 4. 函数执行完毕,内部变量被清空
function calculate() {
    let result = 100
    console.log(result)
}
calculate()  // result被清空
```

**总结**:
```
✅ 函数内部声明的变量,外部无法访问
✅ 函数参数也是局部变量
✅ 不同函数内部变量互不影响
✅ 函数执行完毕后,内部变量被清空
```

#### 1.2.2 块作用域

**定义**: 使用 `{}` 包裹的代码块,内部用let/const声明的变量外部无法访问

```javascript
// 示例1: 基本代码块
{
    let age = 18
    console.log(age)  // 正常
}
// console.log(age)  // 报错

// 示例2: if语句块作用域
if (true) {
    let str = 'hello'
    console.log(str)  // 正常
}
// console.log(str)  // 报错

// 示例3: for循环块作用域
for (let i = 1; i <= 5; i++) {
    console.log(i)  // 正常
}
// console.log(i)  // 报错
```

**let vs var**:

```javascript
// let产生块作用域
{
    let a = 10
}
// console.log(a)  // 报错

// var不产生块作用域
{
    var b = 20
}
console.log(b)  // 20 (能访问)

// for循环中的区别
for (let i = 0; i < 3; i++) {
    setTimeout(() => console.log(i))
}
// 输出: 0 1 2

for (var j = 0; j < 3; j++) {
    setTimeout(() => console.log(j))
}
// 输出: 3 3 3
```

**const常量**:

```javascript
// 1. 常量必须有值
const version = '1.0.0'

// 2. 不能重新赋值
// version = '1.0.1'  // 报错

// 3. 常量值为对象时,属性可以修改
const user = {
    name: '小明',
    age: 18
}

// ❌ 不能重新赋值整个对象
// user = {}  // 报错

// ✅ 可以修改对象的属性
user.name = '小红'
user.gender = '女'

// 4. 数组常量
const arr = [1, 2, 3]
// arr = [4, 5, 6]  // 报错
arr.push(4)  // 可以修改内容
```

**总结**:
```
✅ let声明的变量会产生块作用域,var不会
✅ const声明的常量也会产生块作用域
✅ 不同代码块之间的变量无法互相访问
✅ 推荐使用let或const,避免使用var
```

### 1.3 全局作用域

**定义**: `<script>` 标签和 `.js` 文件的最外层就是全局作用域

```javascript
// 全局作用域
const name = '小明'

function sayHi() {
    // 局部作用域
    console.log('你好' + name)  // 可以访问全局变量
}

sayHi()

// 块作用域中访问全局
const flag = true
let x = 10

if (flag) {
    let y = 5
    console.log(x + y)  // x是全局的
}
```

**全局变量的创建方式**:

```javascript
// 方式1: 直接在全局声明 ✅推荐
let globalVar1 = 'hello'

// 方式2: 为window对象添加属性 ❌不推荐
window.globalVar2 = 'world'

// 方式3: 函数内未声明直接赋值 ❌强烈不推荐
function test() {
    globalVar3 = 'test'
}
test()
console.log(globalVar3)  // 'test'
```

**总结**:
```
❌ 为window对象添加属性默认也是全局的,不推荐
❌ 函数中未使用关键字声明的变量为全局变量,不推荐
✅ 尽可能少的声明全局变量,防止全局变量污染
```

### 1.4 作用域链 ⭐⭐⭐⭐⭐

**定义**: 嵌套关系的作用域串联起来形成了作用域链

**查找机制**: 从内向外逐级查找

```javascript
// 全局作用域
let a = 1
let b = 2

// 局部作用域
function f() {
    let c = 3
    // let a = 10  // 如果声明,则优先使用
    console.log(a)  // 1 (找全局)
    // console.log(d)  // 报错(找不到)
    
    // 嵌套的局部作用域
    function g() {
        let d = 'yo'
        // let b = 20  // 如果声明,则优先使用
        console.log(b)  // 2 (找全局)
        console.log(c)  // 3 (找父级)
    }
    
    g()
}

// 外部无法访问内部
// console.log(c)  // 报错
// console.log(d)  // 报错

f()
```

**查找规则**:

```javascript
// 逐级查找示例
let num1 = 10  // 全局

function outer() {
    let num2 = 20  // outer作用域
    
    function inner() {
        let num3 = 30  // inner作用域
        
        console.log(num3)  // 30 (自己的)
        console.log(num2)  // 20 (父级outer的)
        console.log(num1)  // 10 (全局的)
    }
    
    inner()
}

outer()
```

**总结**:
```
✅ 嵌套关系的作用域串联起来形成作用域链
✅ 相同作用域链中按着从小到大的规则查找变量
✅ 子作用域能够访问父作用域,父级无法访问子级
```

**查找口诀**:
```
先找自己,找不到找爸爸
爸爸没有找爷爷,一直找到全局家
全局还是找不到,程序报错就停下
```

### 1.5 闭包 ⭐⭐⭐⭐⭐

#### 1.5.1 闭包的概念

**定义**: 闭包 = 内层函数 + 外层函数的变量

```javascript
// 最简单的闭包
function outer() {
    const a = 1
    function inner() {
        console.log(a)  // 访问外层变量
    }
    inner()
}
outer()  // 1
```

**闭包的标准写法**:

```javascript
// 闭包: 将内层函数作为返回值
function outer() {
    let count = 1
    
    function inner() {
        count++
        console.log(`函数被调用${count}次`)
    }
    
    return inner
}

const fn = outer()
fn()  // 函数被调用2次
fn()  // 函数被调用3次
fn()  // 函数被调用4次
```

#### 1.5.2 闭包的作用

**作用1: 数据私有化**

```javascript
// ❌ 问题: 全局变量容易被污染
let count = 1
function fn() {
    count++
    console.log(`调用${count}次`)
}
fn()
count = 100  // 外部可以修改

// ✅ 解决: 使用闭包保护数据
function createCounter() {
    let count = 1
    return function() {
        count++
        console.log(`调用${count}次`)
    }
}

const counter = createCounter()
counter()  // 调用2次
counter()  // 调用3次
// count无法从外部访问
```

**作用2: 延长变量生命周期**

```javascript
// 普通函数: 执行完变量被销毁
function normal() {
    let num = 10
    console.log(num)
}
normal()  // 10
normal()  // 10 (每次都是新的)

// 闭包: 变量一直保存在内存中
function createClosure() {
    let num = 10
    return function() {
        num++
        console.log(num)
    }
}

const closure = createClosure()
closure()  // 11
closure()  // 12
closure()  // 13
```

**作用3: 创建私有方法**

```javascript
function createPerson(name) {
    let age = 0  // 私有变量
    
    return {
        getName: function() {
            return name
        },
        getAge: function() {
            return age
        },
        setAge: function(newAge) {
            if (newAge > 0 && newAge < 150) {
                age = newAge
            }
        },
        birthday: function() {
            age++
        }
    }
}

const person = createPerson('小明')
console.log(person.getName())  // 小明
person.setAge(18)
console.log(person.getAge())   // 18
person.birthday()
console.log(person.getAge())   // 19
// console.log(person.age)  // undefined (无法直接访问)
```

#### 1.5.3 闭包的问题

**问题: 内存泄漏**

```javascript
// ❌ 可能导致内存泄漏
function createLargeClosure() {
    const largeData = new Array(1000000).fill('data')
    
    return function() {
        console.log(largeData[0])
    }
}

const closure = createLargeClosure()
// largeData一直保存在内存中

// ✅ 解决: 及时释放
closure = null
```

**总结**:
```
闭包定义: 内层函数 + 外层函数的变量
闭包作用:
  ✅ 封闭数据,实现数据私有
  ✅ 延长变量生命周期
  ✅ 创建私有方法和变量
闭包问题: 
  ❌ 可能造成内存泄漏
```

### 1.6 变量提升

**定义**: var声明的变量可以在声明之前被访问

```javascript
// var的变量提升
console.log(str)  // undefined (不报错)
var str = 'hello'

// 等价于:
var str
console.log(str)  // undefined
str = 'hello'
```

**let/const没有变量提升**:

```javascript
// ❌ let
console.log(name)  // 报错
let name = '小明'

// ❌ const
console.log(age)  // 报错
const age = 18
```

**总结**:
```
❌ var存在变量提升,声明前访问值为undefined
✅ let/const不存在变量提升,推荐使用
✅ 实际开发中推荐先声明再访问变量
```

## 二、函数进阶

### 2.1 函数提升

**定义**: 函数声明可以在声明之前被调用

```javascript
// ✅ 函数声明存在提升
foo()  // '声明之前即被调用...'

function foo() {
    console.log('声明之前即被调用...')
}

// ❌ 函数表达式不存在提升
bar()  // 报错

var bar = function() {
    console.log('函数表达式不存在提升...')
}
```

**总结**:
```
✅ 函数声明存在提升
❌ 函数表达式不存在提升
```

### 2.2 函数参数

#### 2.2.1 默认值

```javascript
// 设置参数默认值
function sayHi(name = '小明', age = 18) {
    console.log(`大家好,我叫${name},今年${age}岁`)
}

sayHi()              // 大家好,我叫小明,今年18岁
sayHi('小红')        // 大家好,我叫小红,今年18岁
sayHi('小刚', 21)    // 大家好,我叫小刚,今年21岁
```

**应用场景**:

```javascript
// 配置对象
function ajax(url, method = 'GET', timeout = 3000) {
    console.log(`${method} ${url} timeout: ${timeout}ms`)
}

ajax('/api/users')                    // GET /api/users timeout: 3000ms
ajax('/api/users', 'POST')            // POST /api/users timeout: 3000ms
ajax('/api/users', 'DELETE', 5000)    // DELETE /api/users timeout: 5000ms
```

**总结**:
```
✅ 声明函数时为形参赋值即为默认值
✅ 未传参时使用默认值
✅ 默认值为undefined时,参数的默认值生效
```

#### 2.2.2 动态参数 (arguments)

```javascript
// arguments是函数内部内置的伪数组
function sum() {
    console.log(arguments)  // [5, 10]
    
    let s = 0
    for (let i = 0; i < arguments.length; i++) {
        s += arguments[i]
    }
    console.log(s)
}

sum(5, 10)       // 15
sum(1, 2, 4)     // 7
sum(1, 2, 3, 4, 5)  // 15
```

**特点**:

```javascript
function test() {
    console.log(arguments.length)        // 有length属性
    console.log(arguments[0])            // 可通过索引访问
    console.log(Array.isArray(arguments))  // false (伪数组)
    
    // 转换为真数组
    const arr = Array.from(arguments)
    const arr2 = [...arguments]
}

test(1, 2, 3)
```

**总结**:
```
✅ arguments是伪数组
✅ arguments动态获取所有实参
❌ 只在function声明的函数中有效
```

#### 2.2.3 剩余参数 ⭐⭐⭐⭐⭐

**语法**: `...变量名`

```javascript
function config(baseURL, ...other) {
    console.log(baseURL)  // 'http://baidu.com'
    console.log(other)    // ['get', 'json']
}

config('http://baidu.com', 'get', 'json')
```

**剩余参数 vs arguments**:

```javascript
// arguments - 伪数组
function fn1() {
    console.log(arguments)
    console.log(Array.isArray(arguments))  // false
}

// 剩余参数 - 真数组
function fn2(...args) {
    console.log(args)
    console.log(Array.isArray(args))  // true
    args.forEach(item => console.log(item))
}

fn1(1, 2, 3)
fn2(1, 2, 3)
```

**应用示例**:

```javascript
// 求和函数
function sum(...numbers) {
    return numbers.reduce((total, num) => total + num, 0)
}

console.log(sum(1, 2, 3))        // 6
console.log(sum(1, 2, 3, 4, 5))  // 15
```

**总结**:
```
✅ ...是语法符号,置于最末形参之前
✅ 获取的剩余实参是真数组
✅ 比arguments更现代,推荐使用
```

### 2.3 箭头函数 ⭐⭐⭐⭐⭐

#### 2.3.1 基本语法

```javascript
// 普通函数
const fn1 = function() {
    console.log('普通函数')
}

// 箭头函数
const fn2 = () => {
    console.log('箭头函数')
}
```

**语法规则**:

```javascript
// 1. 基本语法
const fn1 = () => {
    console.log('hello')
}

// 2. 有参数
const fn2 = (x, y) => {
    console.log(x + y)
}

// 3. 只有一个参数,可以省略小括号
const fn3 = x => {
    console.log(x)
}

// 4. 只有一行代码,可以省略大括号
const fn4 = x => console.log(x)

// 5. 只有一行代码,会自动return
const fn5 = x => x + x
console.log(fn5(5))  // 10

// 6. 返回对象需要加括号
const fn6 = name => ({ name: name })
console.log(fn6('小明'))  // {name: '小明'}
```

**总结**:
```
✅ 箭头函数属于表达式函数,不存在函数提升
✅ 一个参数可以省略()
✅ 一行代码可以省略{},并自动return
✅ 返回对象需要用()包裹
```

#### 2.3.2 箭头函数参数

```javascript
// ❌ 箭头函数没有arguments
const fn1 = () => {
    console.log(arguments)  // 报错
}

// ✅ 使用剩余参数
const fn2 = (...args) => {
    console.log(args)  // 真数组
    return args.reduce((sum, num) => sum + num, 0)
}

console.log(fn2(1, 2, 3, 4, 5))  // 15
```

#### 2.3.3 箭头函数的this ⭐⭐⭐⭐⭐

**普通函数的this**:

```javascript
// 全局函数
console.log(this)  // window

function fn() {
    console.log(this)  // window
}

// 对象方法
const obj = {
    name: 'andy',
    sayHi: function() {
        console.log(this)  // obj
    }
}
obj.sayHi()
```

**箭头函数的this**:

```javascript
// 箭头函数的this是上一层作用域的this

// 1. 全局箭头函数
const fn = () => {
    console.log(this)  // window
}

// 2. 对象方法中的箭头函数
const obj1 = {
    name: 'pink老师',
    sayHi: () => {
        console.log(this)  // window (不是obj)
    }
}

// 3. 正确使用
const obj2 = {
    name: 'pink老师',
    sayHi: function() {
        console.log(this)  // obj2
        
        const inner = () => {
            console.log(this)  // obj2 (继承外层)
        }
        inner()
    }
}
obj2.sayHi()
```

**使用建议**:

```javascript
// ✅ 适合使用箭头函数
arr.forEach(item => console.log(item))
setTimeout(() => console.log('延时'), 1000)
fetch('/api').then(res => res.json())

// ❌ 不适合使用箭头函数
const obj = {
    name: '小明',
    // sayHi: () => {
    //     console.log(this.name)  // undefined
    // }
    sayHi: function() {
        console.log(this.name)  // 小明
    }
}
```

**总结**:
```
箭头函数的this:
  ✅ 没有自己的this
  ✅ this是上一层作用域的this
  ✅ 适合: 数组方法、定时器、Promise
  ❌ 不适合: 对象方法、事件处理函数
```

## 三、解构赋值

### 3.1 数组解构

**定义**: 将数组的单元值快速批量赋值给一系列变量

```javascript
// 普通方式
let arr = [1, 2, 3]
let a = arr[0]
let b = arr[1]
let c = arr[2]

// 数组解构
let [x, y, z] = [1, 2, 3]
console.log(x, y, z)  // 1 2 3
```

**解构细节**:

```javascript
// 1. 变量多于值
let [a, b, c, d] = [1, 2, 3]
console.log(d)  // undefined

// 2. 变量少于值
let [x, y] = [1, 2, 3, 4, 5]
console.log(x, y)  // 1 2

// 3. 使用剩余参数
let [first, ...rest] = [1, 2, 3, 4, 5]
console.log(first)  // 1
console.log(rest)   // [2, 3, 4, 5]

// 4. 默认值
let [a1 = 10, b1 = 20] = [1]
console.log(a1, b1)  // 1 20

// 5. 按需获取
let [, , third] = [1, 2, 3, 4, 5]
console.log(third)  // 3

// 6. 交换变量
let a = 1, b = 2
;[a, b] = [b, a]
console.log(a, b)  // 2 1
```

**总结**:
```
✅ 变量顺序对应数组单元值位置
✅ 变量多于值时,多余为undefined
✅ 变量少于值时,可用...获取剩余
✅ 允许设置默认值
```

### 3.2 对象解构 ⭐⭐⭐⭐⭐

**基本使用**:

```javascript
// 普通方式
const user = {
    name: '小明',
    age: 18
}
let name = user.name
let age = user.age

// 对象解构
const { name, age } = user
console.log(name, age)  // 小明 18
```

**解构细节**:

```javascript
// 1. 属性名必须匹配
const { name, age, gender } = { name: '小明', age: 18 }
console.log(gender)  // undefined

// 2. 重命名
const { name: userName, age: userAge } = { name: '小红', age: 20 }
console.log(userName, userAge)  // 小红 20

// 3. 默认值
const { name = '游客', age = 0 } = { name: '小刚' }
console.log(name, age)  // 小刚 0

// 4. 默认值+重命名
const { name: n = '匿名', age: a = 0 } = { age: 25 }
console.log(n, a)  // 匿名 25
```

**多级对象解构**:

```javascript
const person = {
    name: '小明',
    info: {
        age: 18,
        address: {
            city: '北京',
            street: '朝阳区'
        }
    }
}

const {
    name,
    info: {
        age,
        address: { city, street }
    }
} = person

console.log(name, age, city, street)  // 小明 18 北京 朝阳区
```

**函数参数解构**:

```javascript
// 解构后端数据
const msg = {
    code: 200,
    data: [1, 2, 3],
    message: 'success'
}

function render({ data }) {
    console.log(data)  // [1, 2, 3]

render(msg)

show(msg)
```

**总结**:
```
✅ 对象属性值赋给同名变量
✅ 找不到对应属性时值为undefined
✅ 允许设置默认值和重命名
✅ 支持多级嵌套解构
```

# Day 2: 面向对象与内置构造函数

## 一、深入对象

### 1.1 构造函数

**定义**: 专门用于创建对象的函数,使用new关键字调用

```javascript
// 定义构造函数
function Person(name, age) {
    this.name = name
    this.age = age
}

// 调用函数
const user = new Person('小明', 18)
console.log(user)  // Person {name: '小明', age: 18}
```

**构造函数的特点**:

```javascript
// 1. 使用new关键字调用
function User(name) {
    this.name = name
}
const user = new User('小明')

// 2. 没有参数可以省略() (不推荐)
const user2 = new User

// 3. 返回值是新创建的对象
const user3 = new User('小红')
console.log(user3)  // User {name: '小红'}

// 4. return返回的值无效(返回对象除外)
function Test() {
    this.name = 'test'
    return 100  // 无效
}
const t = new Test()
console.log(t)  // Test {name: 'test'}

// 返回对象时有效
function Test2() {
    this.name = 'test'
    return { age: 18 }  // 有效
}
const t2 = new Test2()
console.log(t2)  // { age: 18 }
```

**总结**:
```
✅ 使用new关键字调用的行为称为实例化
✅ 实例化时没有参数可以省略()
✅ 返回值即为新创建的对象
✅ return返回的值无效(返回对象除外)
✅ 习惯将构造函数首字母大写
```

### 1.2 实例成员

**定义**: 通过构造函数创建的对象称为实例对象,实例对象中的属性和方法称为实例成员

```javascript
// 构造函数
function Person(name, age) {
    // this就是实例对象
    // 实例属性
    this.name = name
    this.age = age
    
    // 实例方法
    this.sayHi = function() {
        console.log(`大家好,我是${this.name}`)
    }
}

// 创建实例对象
const p1 = new Person('小明', 18)
console.log(p1.name)  // 小明
p1.sayHi()            // 大家好,我是小明

// 创建另一个实例
const p2 = new Person('小红', 20)
console.log(p2.name)  // 小红
p2.sayHi()            // 大家好,我是小红
```

**实例独立性**:

```javascript
function Person(name) {
    this.name = name
    this.friends = []
}

const p1 = new Person('小明')
const p2 = new Person('小红')

p1.friends.push('小刚')

console.log(p1.friends)  // ['小刚']
console.log(p2.friends)  // [] (互不影响)
```

**总结**:
```
✅ 构造函数内部this就是实例对象
✅ 为其动态添加的属性和方法即为实例成员
✅ 为构造函数传入参数,动态创建对象
✅ 实例对象彼此独立互不影响
```

### 1.3 静态成员

**定义**: 构造函数本身的属性和方法称为静态成员

```javascript
// 构造函数
function Person(name, age) {
    this.name = name
    this.age = age
}

// 静态属性
Person.eyes = 2
Person.arms = 2

// 静态方法
Person.walk = function() {
    console.log('人都会走路...')
    console.log(this.eyes)  // this指向Person
}

// 调用静态成员
console.log(Person.eyes)  // 2
Person.walk()  // 人都会走路... 2

// 实例无法访问静态成员
const p = new Person('小明', 18)
console.log(p.eyes)  // undefined
// p.walk()  // 报错
```

**应用示例**:

```javascript
function Student(name, age) {
    this.name = name
    this.age = age
    Student.count++
}

// 静态属性: 学生总数
Student.count = 0

// 静态方法
Student.getCount = function() {
    return this.count
}

// 创建学生
const s1 = new Student('小明', 18)
const s2 = new Student('小红', 19)
const s3 = new Student('小刚', 20)

console.log(Student.getCount())  // 3
```

**总结**:
```
✅ 静态成员指添加到构造函数本身的属性和方法
✅ 一般公共特征设置为静态成员
✅ 静态成员方法中的this指向构造函数本身
```

```
函数提升:

参数:
  ✅ 默认值: function(a = 10) {}
  ❌ arguments: 伪数组,动态参数(旧)
```

## 四、常见错误总结

### 4.1 作用域相关

```javascript
// ❌ 错误1: 访问不到变量
function test() {
    let a = 10
}
console.log(a)  // 报错

// ✅ 正确
let a = 10
function test() {
    console.log(a)  // 可以访问
}

// ❌ 错误2: 闭包内存泄漏

    let data = new Array(1000000)
    return function() {
        console.log(data)
    }
}
const fn = create()  // data一直占用内存

// ✅ 正确: 及时释放
fn = null
```

### 4.2 函数相关

```javascript

    name: '小明',
    sayHi: () => {
        console.log(this.name)  // undefined
    }
}

// ✅ 正确
const obj = {
    name: '小明',
    sayHi: function() {
        console.log(this.name)  // 小明
    }
}

// ❌ 错误2: 剩余参数位置错误
function fn(...rest, a) {}  // 语法错误

// ✅ 正确
function fn(a, ...rest) {}
```

### 4.3 数组方法相关

```javascript
// ❌ 错误1: forEach返回值
const arr = [1, 2, 3]
const result = arr.forEach(item => item * 2)
console.log(result)  // undefined

// ✅ 正确: 使用map
const result = arr.map(item => item * 2)
console.log(result)  // [2, 4, 6]

// ❌ 错误2: sort排序数字
const arr = [1, 10, 2, 20]
arr.sort()
console.log(arr)  // [1, 10, 2, 20] (错误)

// ✅ 正确

console.log(arr)  // [1, 2, 10, 20]
```

## 七、开发技巧

### 7.1 闭包应用

```javascript
// 技巧1: 模块化
const module = (function() {
    let private = 'private'
    
    return {
        getPrivate() {
            return private
        }
    }
})()

// 技巧2: 防抖
function debounce(fn, delay) {
    let timer = null
    return function(...args) {
        clearTimeout(timer)
        timer = setTimeout(() => fn.apply(this, args), delay)
    }
}

// 技巧3: 节流
function throttle(fn, delay) {
    let last = 0
    return function(...args) {
        const now = Date.now()
        if (now - last >= delay) {
            fn.apply(this, args)
            last = now
        }
    }
}
```
