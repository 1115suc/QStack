## 1. 网络请求封装

UniApp 不能使用 axios（小程序环境不支持 XMLHttpRequest），需要用 `uni.request` 或基于其封装的库。

### 1.1 为什么不能直接用 axios？

- 微信小程序等环境没有 `XMLHttpRequest`，axios 依赖此对象
- 必须使用 `uni.request` 或 `wx.request` 等原生请求 API
- H5 环境可以用 axios，但为了跨端统一，建议封装 `uni.request`

### 1.2 推荐方案：luch-request（类 axios 的 uni 请求库）

```bash
# 安装（或从插件市场导入）
npm install luch-request
```

### 1.3 封装 uni.request（手动封装，推荐理解原理）

```typescript
// src/utils/request.ts

interface RequestOptions {
  url: string
  method?: 'GET' | 'POST' | 'PUT' | 'DELETE'
  data?: Record<string, any>
  header?: Record<string, string>
  showLoading?: boolean
}

interface ApiResponse<T = any> {
  code: number
  data: T
  message: string
}

// 基础配置
const BASE_URL = import.meta.env.VITE_BASE_URL ?? 'https://api.example.com'
const TIMEOUT = 10000

// 请求拦截逻辑
function getToken(): string {
  return uni.getStorageSync('token') ?? ''
}

// 核心请求函数
function request<T = any>(options: RequestOptions): Promise<T> {
  const { url, method = 'GET', data, header = {}, showLoading = true } = options

  if (showLoading) {
    uni.showLoading({ title: '加载中...', mask: true })
  }

  return new Promise((resolve, reject) => {
    uni.request({
      url: BASE_URL + url,
      method,
      data,
      timeout: TIMEOUT,
      header: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${getToken()}`,
        ...header,
      },
      success(res) {
        const response = res.data as ApiResponse<T>

        // HTTP 状态码处理
        if (res.statusCode === 401) {
          uni.showToast({ title: '登录已过期', icon: 'none' })
          // 跳转登录页
          uni.reLaunch({ url: '/pages/login/index' })
          reject(new Error('Unauthorized'))
          return
        }

        if (res.statusCode !== 200) {
          uni.showToast({ title: `请求失败 ${res.statusCode}`, icon: 'none' })
          reject(new Error(`HTTP Error: ${res.statusCode}`))
          return
        }

        // 业务状态码处理
        if (response.code === 0 || response.code === 200) {
          resolve(response.data)
        } else {
          uni.showToast({ title: response.message || '请求失败', icon: 'none' })
          reject(new Error(response.message))
        }
      },
      fail(err) {
        uni.showToast({ title: '网络异常，请检查网络', icon: 'none' })
        reject(err)
      },
      complete() {
        if (showLoading) {
          uni.hideLoading()
        }
      },
    })
  })
}

// 快捷方法
export const http = {
  get<T = any>(url: string, data?: Record<string, any>, options?: Partial<RequestOptions>) {
    return request<T>({ url, method: 'GET', data, ...options })
  },
  post<T = any>(url: string, data?: Record<string, any>, options?: Partial<RequestOptions>) {
    return request<T>({ url, method: 'POST', data, ...options })
  },
  put<T = any>(url: string, data?: Record<string, any>, options?: Partial<RequestOptions>) {
    return request<T>({ url, method: 'PUT', data, ...options })
  },
  delete<T = any>(url: string, data?: Record<string, any>, options?: Partial<RequestOptions>) {
    return request<T>({ url, method: 'DELETE', data, ...options })
  },
}
```

### 2.2 用户状态 Store

```typescript
// src/stores/user.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { userApi, type UserInfo } from '@/api/user'

export const useUserStore = defineStore(
  'user',
  () => {
    // State
    const token = ref<string>('')
    const userInfo = ref<UserInfo | null>(null)

    // Getters
    const isLogin = computed(() => !!token.value)
    const nickname = computed(() => userInfo.value?.nickname ?? '游客')

    // Actions
    async function login(code: string) {
      const res = await userApi.login({ code })
      token.value = res.token
      userInfo.value = res.userInfo
      // 同步到本地存储
      uni.setStorageSync('token', res.token)
    }

    async function fetchUserInfo() {
      if (!isLogin.value) return
      userInfo.value = await userApi.getUserInfo()
    }

    function logout() {
      token.value = ''
      userInfo.value = null
      uni.removeStorageSync('token')
      uni.reLaunch({ url: '/pages/login/index' })
    }

    return { token, userInfo, isLogin, nickname, login, fetchUserInfo, logout }
  },
  {
    // pinia-plugin-persistedstate 持久化配置
    persist: {
      key: 'user-store',
      storage: {
        // UniApp 使用 uni.setStorageSync 持久化
        getItem: (key) => uni.getStorageSync(key),
        setItem: (key, value) => uni.setStorageSync(key, value),
      },
    },
  }
)
```

## 4. Composables 组合式函数

### 4.1 请求状态封装 useRequest

```typescript
// src/composables/useRequest.ts
import { ref } from 'vue'

interface UseRequestOptions<T> {
  immediate?: boolean         // 是否立即执行
  initialData?: T             // 初始数据
  onSuccess?: (data: T) => void
  onError?: (err: Error) => void
}

export function useRequest<T>(
  apiFn: (...args: any[]) => Promise<T>,
  options: UseRequestOptions<T> = {}
) {
  const { immediate = false, initialData, onSuccess, onError } = options

  const data = ref<T | undefined>(initialData) as Ref<T>
  const loading = ref(false)
  const error = ref<Error | null>(null)

  async function run(...args: any[]) {
    loading.value = true
    error.value = null
    try {
      data.value = await apiFn(...args)
      onSuccess?.(data.value)
    } catch (err) {
      error.value = err as Error
      onError?.(err as Error)
    } finally {
      loading.value = false
    }
  }

  if (immediate) run()

  return { data, loading, error, run }
}
```

```typescript
// 使用示例
import { useRequest } from '@/composables/useRequest'
import { userApi } from '@/api/user'

const { data: userInfo, loading, run: fetchUser } = useRequest(
  userApi.getUserInfo,
  {
    immediate: false,
    onSuccess(data) {
      console.log('获取成功', data)
    },
  }
)

onLoad(() => fetchUser())
```

### 4.2 分页列表 usePagination

```typescript
// src/composables/usePagination.ts
import { ref } from 'vue'

interface PaginationOptions<T> {
  apiFn: (params: { page: number; pageSize: number; [key: string]: any }) => Promise<{
    list: T[]
    total: number
  }>
  pageSize?: number
}

export function usePagination<T>(options: PaginationOptions<T>) {
  const { apiFn, pageSize = 10 } = options

  const list = ref<T[]>([]) as Ref<T[]>
  const page = ref(1)
  const total = ref(0)
  const loading = ref(false)
  const finished = ref(false)   // 是否加载完毕

  // 初始加载 / 刷新
  async function refresh(extraParams = {}) {
    page.value = 1
    finished.value = false
    loading.value = true
    try {
      const res = await apiFn({ page: 1, pageSize, ...extraParams })
      list.value = res.list
      total.value = res.total
      if (res.list.length < pageSize) finished.value = true
    } finally {
      loading.value = false
      uni.stopPullDownRefresh()
    }
  }

  // 加载更多
  async function loadMore(extraParams = {}) {
    if (loading.value || finished.value) return
    loading.value = true
    try {
      const nextPage = page.value + 1
      const res = await apiFn({ page: nextPage, pageSize, ...extraParams })
      list.value = [...list.value, ...res.list]
      page.value = nextPage
      if (res.list.length < pageSize) finished.value = true
    } finally {
      loading.value = false
    }
  }

  return { list, loading, finished, total, refresh, loadMore }
}
```

```typescript
// 使用示例
import { usePagination } from '@/composables/usePagination'
import { articleApi } from '@/api/article'

const { list, loading, finished, refresh, loadMore } = usePagination({
  apiFn: articleApi.getList,
  pageSize: 10,
})

onLoad(() => refresh())
onPullDownRefresh(() => refresh())
onReachBottom(() => loadMore())
```

### 4.3 本地存储封装 useStorage

```typescript
// src/composables/useStorage.ts
import { ref, watch } from 'vue'

export function useStorage<T>(key: string, defaultValue: T) {
  const storedValue = uni.getStorageSync(key)
  const data = ref<T>(storedValue !== '' ? storedValue : defaultValue)

  watch(
    data,
    (newVal) => {
      if (newVal === null || newVal === undefined) {
        uni.removeStorageSync(key)
      } else {
        uni.setStorageSync(key, newVal)
      }
    },
    { deep: true }
  )

  function remove() {
    uni.removeStorageSync(key)
    data.value = defaultValue
  }

  return { data, remove }
}
```

## 5. 条件编译

条件编译是 UniApp 的特色功能，允许针对特定平台写代码。

### 5.1 语法规则

```
#ifdef  平台名     → 仅在该平台编译
#ifndef 平台名     → 除该平台外编译
#endif             → 结束条件块
```

**平台标识符：**

| 标识符 | 说明 |
|--------|------|
| `H5` | H5 网页 |
| `MP-WEIXIN` | 微信小程序 |
| `MP-ALIPAY` | 支付宝小程序 |
| `APP-PLUS` | App（Android + iOS）|
| `APP-ANDROID` | 仅 Android |
| `APP-IOS` | 仅 iOS |
| `MP` | 所有小程序 |

### 5.2 在 Vue 模板中使用

```html
<template>
  <view>
    <!-- #ifdef MP-WEIXIN -->
    <button open-type="getUserInfo" @getuserinfo="getWxUserInfo">
      微信授权登录
    </button>
    <!-- #endif -->

    <!-- #ifdef H5 -->
    <button @tap="h5Login">H5 登录</button>
    <!-- #endif -->

    <!-- #ifdef APP-PLUS -->
    <button @tap="appLogin">App 登录</button>
    <!-- #endif -->
  </view>
</template>
```
